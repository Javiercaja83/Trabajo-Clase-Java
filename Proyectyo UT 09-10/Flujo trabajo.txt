# 1. Ir a la carpeta del proyecto
cd zion

# 2. Actualizar siempre primero (MUY IMPORTANTE)
git pull origin main

# 3. (Opcional pero recomendado) Crear una rama para su trabajo
git checkout -b nombre-de-la-rama   # ejemplo: git checkout -b feature/usuarios

# 4. Trabajar (hacer cambios en el código)

# 5. Cuando termine de trabajar:
git add .                     # o git add nombre-archivo.java
git commit -m "Mensaje claro de lo que hizo"
git push origin nombre-de-la-rama     # si está en una rama

# Si está trabajando directamente en main (no recomendado pero simple):
git push origin main
